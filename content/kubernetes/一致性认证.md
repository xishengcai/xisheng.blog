# install test tool sonobuoy
[download release page](https://github.com/vmware-tanzu/sonobuoy/releases)
```
# download binar
wget https://github.com/vmware-tanzu/sonobuoy/releases/download/v0.18.0/sonobuoy_0.18.0_linux_amd64.tar.gz 
tar -zxvf sonobuoy_0.18.0_linux_amd64.tar.gz 
mv sonobuoy /usr/local/bin/

# run test
sonobuoy run --mode=certified-conformance 

# query status
sonobuoy status --json | jq
![一致性测试进度查询](https:/xisheng.vip/images/cncf-conformance.jpg)

# output test result
outfile=$(sonobuoy retrieve)
mkdir ./results; tar xzf $outfile -C ./results

# copy result to local
scp  root@xxxxx:/root/results/plugins/e2e/results/global/* 

```
